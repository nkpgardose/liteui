A base component that styled up button element with necessary props on it.

**Sizes**

```js
<Btn size="xsm">Ok</Btn>
<Btn size="sm">Submit</Btn>
<Btn size="lg">Submit</Btn>
<Btn size="xlg">Submit</Btn>
```

**Modifiers**

```js
<Btn>Submit</Btn>
<Btn disabled>Submit</Btn>
<Btn modifiers="primary">Submit</Btn>
<Btn modifiers="primary" disabled>Submit</Btn>
<Btn modifiers="accent">Submit</Btn>
<Btn modifiers="accent" disabled>Submit</Btn>
```

**Misc**

```js
<Btn size="xsm">Ok</Btn>
<Btn size="sm">Submit</Btn>
<Btn>Submit</Btn>
<Btn size="lg">Submit</Btn>
<Btn size="xlg">Submit</Btn>
<Btn disabled>Submit</Btn>
```

```js
<Btn  modifiers="primary" size="xsm">Ok</Btn>
<Btn  modifiers="primary" size="sm">Submit</Btn>
<Btn modifiers="primary" >Submit</Btn>
<Btn  modifiers="primary" size="lg">Submit</Btn>
<Btn  modifiers="primary" size="xlg">Submit</Btn>
<Btn  modifiers="primary" disabled>Submit</Btn>
```

```js
<Btn  modifiers="accent" size="xsm">Ok</Btn>
<Btn  modifiers="accent" size="sm">Submit</Btn>
<Btn modifiers="accent" >Submit</Btn>
<Btn  modifiers="accent" size="lg">Submit</Btn>
<Btn  modifiers="accent" size="xlg">Submit</Btn>
<Btn  modifiers="accent" disabled>Submit</Btn>
```
